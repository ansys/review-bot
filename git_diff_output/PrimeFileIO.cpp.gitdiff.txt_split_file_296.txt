+void PrimeFileIO::ReadToken(FILE *f, char *token)
+{
+    int i;
+    while ((i = getc(f)) != EOF)
+    {
+        switch ((char)i)
+        {
+        case '(':
+        case ')':
+            *token = '\0';
+            ungetc((char)i, f);
+            return;
+        case ' ':
+            *token = '\0';
+            return;
+        default:
+            *token = (char)i;
+            token++;
+            break;
+        }
+    }
+    return;
